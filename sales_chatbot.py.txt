# Function to get chatbot response
def chatbot_response(user_input):
    user_input = user_input.lower()

    if any(word in user_input for word in ["hi", "hello", "hey"]):
        return random.choice(responses["greeting"])
    elif "sale" in user_input:
        return random.choice(responses["sales"])
    elif "profit" in user_input:
        return random.choice(responses["profit"])
    elif "forecast" in user_input or "future" in user_input:
        return random.choice(responses["forecast"])
    elif any(word in user_input for word in ["bye", "exit", "quit"]):
        return random.choice(responses["goodbye"])
    else:
        return random.choice(responses["default"])
